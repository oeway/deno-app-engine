4. Stream Execution

Endpoint: POST /api/kernels/:id/execute/stream

const response = await fetch(`/api/kernels/${kernelId}/execute/stream`, {
    method: 'POST',
    headers: { 'Content-Type': 'application/json' },
    body: JSON.stringify({
        code: `
import asyncio
print("Starting stream test...")
for i in range(5):
    print(f"Processing item {i}...")
    await asyncio.sleep(1)
    if i % 2 == 0:
        print(f"  - Even number found: {i}")
    else:
        print(f"  - Odd number found: {i}")
print("Stream test complete!")`
    })
});
