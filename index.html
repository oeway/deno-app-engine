4. Stream Execution

Endpoints:

POST /api/kernels/:id/execute/submit - Submit code for execution

GET /api/kernels/:id/execute/stream/:session_id - Stream execution results

// First submit the code for execution
const execResponse = await fetch(`/api/kernels/${kernelId}/execute/submit`, {
    method: 'POST',
    headers: { 'Content-Type': 'application/json' },
    body: JSON.stringify({
        code: `
import asyncio
print("Starting stream test...")
for i in range(5):
    print(f"Processing item {i}...")
    await asyncio.sleep(1)
    if i % 2 == 0:
        print(f"  - Even number found: {i}")
    else:
        print(f"  - Odd number found: {i}")
print("Stream test complete!")`
    })
});

const { session_id } = await execResponse.json();

// Then create an EventSource to receive streaming updates
const eventSource = new EventSource(`/api/kernels/${kernelId}/execute/stream/${session_id}`);

eventSource.onmessage = (event) => {
    const data = JSON.parse(event.data);
    
    let formattedOutput = '';
    if (data.type === 'stream') {
        formattedOutput = data.data.text;
    } else if (data.type === 'execute_result') {
        formattedOutput = `Result: ${JSON.stringify(data.data.data)}\n`;
    } else if (data.type === 'error') {
        formattedOutput = `Error: ${data.data.message}\n`;
    } else {
        formattedOutput = `${data.type}: ${JSON.stringify(data.data)}\n`;
    }
    
    console.log(formattedOutput);
};

eventSource.onerror = (error) => {
    console.error('Stream error:', error);
    eventSource.close();
};
